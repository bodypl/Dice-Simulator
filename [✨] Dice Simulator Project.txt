import random # packet to generate random numbers
dice_faces = ["1 [.]", "2 [:]", "3 [:.]", "4 [::]", "5 [::.]", "6 [:::]"]
count = 0 # dice counting starts at zero every time

# spin process
while True:
    ask = input("spin the dice? (Y/n): ")
    
    if ask == "Y":
        dice = random.choice(dice_faces) # get the dice number
        count += 1 # spin counting
        print("result:", dice)
        dice_number = int(dice[0])
        if dice_number % 2 == 0:
            print("type: even number")
        else:
            print("type: odd number")
        print("total spins:", count)
        print(line)
        if count == 100: # activates the achievement system depends on count amount
            print("achievement unlocked: gambling is life!")

            print("achievement unlocked: getting addicted into it")
        elif count == 25:
            print("achievement unlocked: got into gambling")
        elif count == 10:
            print("achievement unlocked: baby steps")
        else:
            continue
        print(line)
    elif ask == "n":
        print("gambling is over!")
        print("total spin amount this round:", count)
        break
    elif ask == "M":
        print("Achievements:")
        if count >= 100:
            print("gambling is life! (100)")
            print("cannot quit now (75)")
            print("getting addicted into it (50)")
            print("got into gambling (25)")
            print("baby steps (10)")

            print("got into gambling (25)")
